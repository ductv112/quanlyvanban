```
import { callFunction, callFunctionOne } from '../lib/db/query.js';

export interface CalendarEventRow {
  id: number; title: string; description: string;
  start_time: string; end_time: string; all_day: boolean;
  color: string; repeat_type: string; scope: string;
  unit_id: number; created_by: number; creator_name: string;
  created_at: string; updated_at: string;
}

interface DbResult { success: boolean; message: string; id?: number; }

export const calendarRepository = {
  getList(scope, unitId, staffId, start, end) → callFunction('public.fn_calendar_event_get_list', [scope, unitId, staffId, start, end]),
  getById(id) → callFunctionOne('public.fn_calendar_event_get_by_id', [id]),
  create(title, description, startTime, endTime, allDay, color, repeatType, scope, unitId, createdBy) → callFunctionOne DbResult,
  update(id, title, description, startTime, endTime, allDay, color, repeatType, scope, unitId, staffId) → callFunctionOne DbResult,
  delete(id, staffId) → callFunctionOne DbResult,
};
```

**calendar.ts route:**
- `GET /events` — query params: scope, start, end. Use AuthRequest staffId/unitId. Returns event list.
- `GET /events/:id` — single event by ID.
- `POST /events` — body: {title, description, start_time, end_time, all_day, color, repeat_type, scope}. Use staffId as created_by. For scope='unit' or 'leader', use requireRoles or check admin. Returns 201.
- `PUT /events/:id` — update event. Body same as create. Returns result.
- `DELETE /events/:id` — soft delete. Returns result.
- All routes wrapped in try/catch with handleDbError.
- Import handleDbError from '../lib/error-handler.js'.

**directory.ts route:**
- `GET /` — query params: unit_id, department_id, search, page (default 1), page_size (default 20). Returns {data: rows, total: rows[0]?.total_count || 0}.
- Wrap in try/catch with handleDbError.
