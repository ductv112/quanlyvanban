<objective>
Implement the BullMQ worker that consumes `poll-sign-status` jobs. This is the heart of the async sign flow — polls provider every 5s, embeds signature when done, uploads signed PDF, updates DB, emits Socket events, creates bell notifications.

Purpose: Decouple the provider-waits (3 minutes worst case) from the user-facing API. When ready, user gets notified via Socket (if online) or bell (next login). User can close browser mid-flow; worker still completes the transaction.

Output: 1 worker file + 1 sign events helper + 1 repository extension + server.ts integration.
</objective>

Create worker file (~280 lines). Single concurrency worker — one sign at a time per queue to avoid provider rate-limiting and simplify state.

```typescript
/**
 * BullMQ Worker: poll-sign-status
 *
 * Job flow (per job instance):
 *   1. Read sign_transactions row by ID
 *   2. Short-circuit if status !== 'pending' (user may have cancelled)
 *   3. Call provider.getSignStatus(providerTxnId)
 *   4. If status='pending' + attempt < MAX_POLL_ATTEMPTS:
 *        incrementRetry, re-enqueue with delay=5s, attempt+1
 *   5. If status='pending' + attempt >= MAX_POLL_ATTEMPTS:
 *        updateStatus('expired'), emit SIGN_FAILED, create bell, cleanup
 *   6. If status='completed':
 *        getPlaceholder → signPdf → uploadFile → finalizeSign → signTransaction.complete
 *        → emit SIGN_COMPLETED, create bell, remove placeholder
 *   7. If status='failed' or 'expired':
 *        updateStatus with same status, emit SIGN_FAILED, create bell, cleanup
 *
 * Concurrency = 1 (serialize to avoid provider throttling, simpler tracing).
 * Can tune later via WORKER_CONCURRENCY env.
 *
 * Graceful shutdown: server.ts calls stopSigningWorker() on SIGTERM —
 * worker.close() waits for current job to finish (up to 30s) then exits.
 */

import { Worker, type Job } from 'bullmq';
import pino from 'pino';
import {
  SIGNING_QUEUE_NAME,
  POLL_SIGN_STATUS,
  POLL_INTERVAL_MS,
  MAX_POLL_ATTEMPTS,
  type PollSignStatusJob,
} from '../lib/queue/types.js';
import { createRedisConnection } from '../lib/queue/redis-connection.js';
import { enqueuePollSignStatus } from '../lib/queue/signing-queue.js';
import { signTransactionRepository, type SignAttachmentType } from '../repositories/sign-transaction.repository.js';
import { staffSigningConfigRepository } from '../repositories/staff-signing-config.repository.js';
import { attachmentSignRepository } from '../repositories/attachment-sign.repository.js';
import { noticeRepository } from '../repositories/notice.repository.js';
import { getProviderByCodeWithCredentials } from '../services/signing/providers/provider-factory.js';
import { signPdf } from '../services/signing/pdf-signer.js';
import { getPlaceholder, removePlaceholder } from '../lib/signing/placeholder-store.js';
import { buildSignedObjectKey } from '../lib/signing/sign-helpers.js';
import { emitSignCompleted, emitSignFailed } from '../lib/signing/sign-events.js';
import { minioClient } from '../lib/minio/client.js';

const BUCKET = process.env.MINIO_BUCKET || 'documents';
const logger = pino({ name: 'signing-worker' });
let worker: Worker<PollSignStatusJob> | null = null;

async function uploadSignedPdf(key: string, buffer: Buffer): Promise<void> {
  await minioClient.putObject(BUCKET, key, buffer, buffer.length, {
    'Content-Type': 'application/pdf',
  });
}

/** Wrapper for all notification + socket emission on failure. */
async function handleFailure(
  txnId: number,
  staffId: number,
  providerCode: string,
  attachmentId: number,
  attachmentType: string,
  status: 'failed' | 'expired',
  errorMessage: string,
): Promise<void> {
  const errMsg = errorMessage.slice(0, 500);
  await signTransactionRepository.updateStatus(txnId, status, errMsg);
  await removePlaceholder(txnId);

  emitSignFailed(staffId, {
    transaction_id: txnId,
    provider_code: providerCode,
    attachment_id: attachmentId,
    attachment_type: attachmentType,
    error_message: errMsg,
    status,
  });

  try {
    await noticeRepository.createForStaff(
      staffId,
      `Ký số thất bại (${status === 'expired' ? 'hết hạn' : 'lỗi'})`,
      `Giao dịch ký số #${txnId} ${status === 'expired' ? 'hết hạn sau 3 phút' : 'thất bại'}: ${errMsg}`,
      'SIGN_RESULT',
    );
  } catch (err) {
    logger.warn({ err, txnId }, 'Failed to create failure notification');
  }
}

async function processJob(job: Job<PollSignStatusJob>): Promise<void> {
  const { signTransactionId, providerTransactionId, attempt } = job.data;
  const ctx = { jobId: job.id, txnId: signTransactionId, attempt };
  logger.info(ctx, 'Processing poll-sign-status job');

  // 1. Read DB row
  const txn = await signTransactionRepository.getById(signTransactionId);
  if (!txn) {
    logger.warn(ctx, 'Transaction not found — skipping');
    return;
  }
  if (txn.status !== 'pending') {
    logger.info({ ...ctx, status: txn.status }, 'Transaction no longer pending — skipping (cancelled/done)');
    await removePlaceholder(signTransactionId);
    return;
  }

  // 2. Load user config for provider API call
  const userConfig = await staffSigningConfigRepository.get(txn.staff_id, txn.provider_code);
  if (!userConfig || !userConfig.is_verified) {
    await handleFailure(
      signTransactionId, txn.staff_id, txn.provider_code,
      txn.attachment_id, txn.attachment_type,
      'failed', 'Cấu hình ký số của người dùng không còn hợp lệ',
    );
    return;
  }

  // 3. Get provider with credentials (may have changed if admin updated config)
  const active = await getProviderByCodeWithCredentials(txn.provider_code as 'SMARTCA_VNPT' | 'MYSIGN_VIETTEL');
  if (!active) {
    await handleFailure(
      signTransactionId, txn.staff_id, txn.provider_code,
      txn.attachment_id, txn.attachment_type,
      'failed', 'Provider đã bị Admin xóa hoặc vô hiệu hóa',
    );
    return;
  }

  // 4. Poll provider status
  let status;
  try {
    status = await active.provider.getSignStatus(
      active.credentials,
      { userId: userConfig.user_id, credentialId: userConfig.credential_id },
      providerTransactionId,
    );
  } catch (err: unknown) {
    const msg = err instanceof Error ? err.message : String(err);
    logger.warn({ ...ctx, err: msg }, 'Provider getSignStatus threw — transient, will retry');
    // Transient failure — count as retry
    await signTransactionRepository.incrementRetry(signTransactionId);
    if (attempt >= MAX_POLL_ATTEMPTS) {
      await handleFailure(
        signTransactionId, txn.staff_id, txn.provider_code,
        txn.attachment_id, txn.attachment_type,
        'expired', `Hết thời gian chờ sau ${MAX_POLL_ATTEMPTS} lần thử (3 phút). Lỗi cuối: ${msg}`,
      );
      return;
    }
    await enqueuePollSignStatus({
      signTransactionId, providerTransactionId,
      placeholderPdfKey: job.data.placeholderPdfKey,
      attempt: attempt + 1,
    }, POLL_INTERVAL_MS);
    return;
  }

  // 5. Branch on provider status
  if (status.status === 'pending') {
    if (attempt >= MAX_POLL_ATTEMPTS) {
      await handleFailure(
        signTransactionId, txn.staff_id, txn.provider_code,
        txn.attachment_id, txn.attachment_type,
        'expired', 'Hết thời gian chờ xác nhận OTP (3 phút)',
      );
      return;
    }
    await signTransactionRepository.incrementRetry(signTransactionId);
    await enqueuePollSignStatus({
      signTransactionId, providerTransactionId,
      placeholderPdfKey: job.data.placeholderPdfKey,
      attempt: attempt + 1,
    }, POLL_INTERVAL_MS);
    logger.debug({ ...ctx, status: 'pending' }, 'Still pending — re-queued');
    return;
  }

  if (status.status === 'failed' || status.status === 'expired') {
    await handleFailure(
      signTransactionId, txn.staff_id, txn.provider_code,
      txn.attachment_id, txn.attachment_type,
      status.status === 'expired' ? 'expired' : 'failed',
      status.errorMessage ?? 'Provider không trả về lý do cụ thể',
    );
    return;
  }

  // 6. Completed — embed signature + upload
  if (status.status !== 'completed' || !status.signatureBase64) {
    await handleFailure(
      signTransactionId, txn.staff_id, txn.provider_code,
      txn.attachment_id, txn.attachment_type,
      'failed', 'Provider trả completed nhưng không có signature_base64',
    );
    return;
  }

  let placeholderBuf: Buffer;
  try {
    placeholderBuf = await getPlaceholder(signTransactionId);
  } catch (err: unknown) {
    await handleFailure(
      signTransactionId, txn.staff_id, txn.provider_code,
      txn.attachment_id, txn.attachment_type,
      'failed', 'Không thể đọc placeholder PDF: ' + (err instanceof Error ? err.message : String(err)),
    );
    return;
  }

  let signed;
  try {
    signed = await signPdf(placeholderBuf, status.signatureBase64);
  } catch (err: unknown) {
    await handleFailure(
      signTransactionId, txn.staff_id, txn.provider_code,
      txn.attachment_id, txn.attachment_type,
      'failed', 'Không thể embed signature vào PDF: ' + (err instanceof Error ? err.message : String(err)),
    );
    return;
  }

  // Build signed object key — need original attachment file_path. Fetch via canSign again? Simpler:
  // infer from placeholder key pattern — attachment file_name basename.
  // Better: query attachment table via canSign (reuse permission check - returns file_path).
  const canSignAgain = await attachmentSignRepository.canSign(
    txn.attachment_id,
    txn.attachment_type as SignAttachmentType,
    txn.staff_id,
  );
  const originalKey = canSignAgain.file_path ?? `unknown/${txn.attachment_id}.pdf`;
  const signedKey = buildSignedObjectKey(originalKey, txn.attachment_type as SignAttachmentType, signTransactionId);

  try {
    await uploadSignedPdf(signedKey, signed.signedPdf);
  } catch (err: unknown) {
    await handleFailure(
      signTransactionId, txn.staff_id, txn.provider_code,
      txn.attachment_id, txn.attachment_type,
      'failed', 'Không thể upload signed PDF: ' + (err instanceof Error ? err.message : String(err)),
    );
    return;
  }

  // DB atomic: complete transaction + finalize attachment
  const completeRes = await signTransactionRepository.complete(
    signTransactionId,
    status.signatureBase64,
    signedKey,
  );
  if (!completeRes.success) {
    logger.error({ ...ctx, msg: completeRes.message }, 'signTransaction.complete failed');
    // Don't bail — attempt finalize anyway
  }

  const finalizeRes = await attachmentSignRepository.finalizeSign({
    attachmentId: txn.attachment_id,
    attachmentType: txn.attachment_type as SignAttachmentType,
    signedFilePath: signedKey,
    signProviderCode: txn.provider_code,
    signTransactionId: signTransactionId,
  });
  if (!finalizeRes.success) {
    logger.warn({ ...ctx, msg: finalizeRes.message }, 'finalize_sign reported failure');
  }

  // Cleanup placeholder
  await removePlaceholder(signTransactionId);

  // Socket + notification
  emitSignCompleted(txn.staff_id, {
    transaction_id: signTransactionId,
    provider_code: txn.provider_code,
    attachment_id: txn.attachment_id,
    attachment_type: txn.attachment_type,
    doc_id: txn.doc_id,
    doc_type: txn.doc_type,
    signed_file_path: signedKey,
    completed_at: new Date().toISOString(),
  });

  try {
    await noticeRepository.createForStaff(
      txn.staff_id,
      'Ký số thành công',
      `Giao dịch ký số #${signTransactionId} (${canSignAgain.file_name ?? 'file'}) đã hoàn tất.`,
      'SIGN_RESULT',
    );
  } catch (err) {
    logger.warn({ err, txnId: signTransactionId }, 'Failed to create completion notification');
  }

  logger.info(ctx, 'Sign completed successfully');
}

export function startSigningWorker(): Worker<PollSignStatusJob> | null {
  if (process.env.WORKER_ENABLED === 'false') {
    logger.info('Signing worker DISABLED by WORKER_ENABLED=false');
    return null;
  }
  if (worker) {
    logger.warn('Signing worker already started');
    return worker;
  }
  worker = new Worker<PollSignStatusJob>(SIGNING_QUEUE_NAME, processJob, {
    connection: createRedisConnection(),
    concurrency: Number(process.env.WORKER_CONCURRENCY) || 1,
    autorun: true,
  });

  worker.on('failed', (job, err) => {
    logger.error({ jobId: job?.id, err: err.message }, 'Worker job failed (exception)');
  });
  worker.on('completed', (job) => {
    logger.debug({ jobId: job.id }, 'Worker job completed');
  });
  worker.on('error', (err) => {
    logger.error({ err: err.message }, 'Worker error event');
  });

  logger.info({ queue: SIGNING_QUEUE_NAME }, 'Signing worker started');
  return worker;
}

export async function stopSigningWorker(timeoutMs: number = 30_000): Promise<void> {
  if (!worker) return;
  logger.info('Stopping signing worker...');
  await Promise.race([
    worker.close(),
    new Promise<void>((resolve) => setTimeout(resolve, timeoutMs)),
  ]);
  worker = null;
  logger.info('Signing worker stopped');
}
```
