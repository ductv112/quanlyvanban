import type { INotificationChannel } from './types.js';
import { MockFcmChannel } from './notification.fcm.mock.js';
import { MockZaloChannel } from './notification.zalo.mock.js';
import { MockSmsChannel } from './notification.sms.mock.js';
import { MockEmailChannel } from './notification.email.mock.js';

export function getNotificationChannel(channel: string): INotificationChannel {
  // When MOCK_EXTERNAL !== 'false', always return mock
  const useMock = process.env.MOCK_EXTERNAL !== 'false';
  switch (channel) {
    case 'fcm': return useMock ? new MockFcmChannel() : new MockFcmChannel();
    case 'zalo': return useMock ? new MockZaloChannel() : new MockZaloChannel();
    case 'sms': return useMock ? new MockSmsChannel() : new MockSmsChannel();
    case 'email': return useMock ? new MockEmailChannel() : new MockEmailChannel();
    default: throw new Error(`Unknown notification channel: ${channel}`);
  }
}
