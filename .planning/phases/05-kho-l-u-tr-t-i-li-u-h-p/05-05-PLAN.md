PUT    /:id              — update contract
DELETE /:id              — delete contract (status=0 only)
GET    /:id/dinh-kem     — list attachments
POST   /:id/dinh-kem     — upload attachment (multipart: file)
DELETE /:id/dinh-kem/:attachmentId — delete attachment
