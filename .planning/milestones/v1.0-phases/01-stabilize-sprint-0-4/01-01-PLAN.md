```typescript
// filterTree — filters tree by keyword match on title
const filterTree = useCallback((nodes: TreeNode[], keyword: string): TreeNode[] => {
  return nodes.reduce((acc, node) => {
    const children = node.children ? filterTree(node.children, keyword) : [];
    const match = node.title.toLowerCase().includes(keyword.toLowerCase());
    if (match || children.length > 0) {
      acc.push({ ...node, children: children.length > 0 ? children : node.children });
    }
    return acc;
  }, [] as TreeNode[]);
}, []);

  return nodes.map((n) => ({

    title: n.title,
    children: n.children ? flattenTreeForSelect(n.children) : undefined,
  }));
}, []);
```

    ```typescript
    import type { TreeNode } from '@/types/tree';

    /**
     * Đệ quy lọc cây theo từ khóa (so khớp title không phân biệt hoa thường).
     * Giữ lại node cha nếu bất kỳ node con nào khớp.
     */
    export function filterTree(nodes: TreeNode[], keyword: string): TreeNode[] {
      if (!keyword.trim()) return nodes;
      return nodes.reduce((acc, node) => {
        const children = node.children ? filterTree(node.children, keyword) : [];
        const match = node.title.toLowerCase().includes(keyword.toLowerCase());
        if (match || children.length > 0) {
          acc.push({ ...node, children: children.length > 0 ? children : node.children });
        }
        return acc;
      }, [] as TreeNode[]);
    }

    /**
     * Chuyển đổi TreeNode[] thành định dạng treeData cho Ant Design TreeSelect.
     * Mỗi node: { value: key, title, children }
     */
    export function flattenTreeForSelect(nodes: TreeNode[]): { value: number; title: string; children?: any[] }[] {
      return nodes.map((n) => ({
        value: n.key,
        title: n.title,
        children: n.children ? flattenTreeForSelect(n.children) : undefined,
      }));
    }
    ```

    Important notes:
    - These are pure functions (not React hooks) — callers use useMemo/useCallback at call sites if needed
    - filterTree handles empty keyword gracefully (returns nodes unchanged)
    - Vietnamese JSDoc comments for the functions
    - Export both as named exports from tree-utils.ts
    - The TreeNode type uses index signature [key: string]: any to allow page-specific fields (is_unit, etc.)
