```sql
-- ============================================================================
-- e-Office Master Schema v2.0 (consolidated 2026-04-22)
-- Hợp nhất từ: 000_full_schema + 7 quick_*.sql + 040-046 (trừ 041 data mig + 043 seed)
-- Idempotent: DROP IF EXISTS trước CREATE cho TẤT CẢ SPs/triggers/indexes
-- Chạy lại lần thứ 2 liên tiếp KHÔNG lỗi
-- ============================================================================

\set ON_ERROR_STOP on

-- ─── Phần 1: Extensions (idempotent CREATE IF NOT EXISTS) ───
CREATE EXTENSION IF NOT EXISTS "uuid-ossp";
CREATE EXTENSION IF NOT EXISTS "pgcrypto";
CREATE EXTENSION IF NOT EXISTS "unaccent";
CREATE EXTENSION IF NOT EXISTS "pg_trgm";

-- ─── Phần 2: Schemas (idempotent) ───
CREATE SCHEMA IF NOT EXISTS edoc;
CREATE SCHEMA IF NOT EXISTS esto;
CREATE SCHEMA IF NOT EXISTS cont;
CREATE SCHEMA IF NOT EXISTS iso;

-- ─── Phần 3: DROP ALL OLD FUNCTIONS (idempotent guard chống SP overload) ───
-- DROP hết SPs fn_* trong 5 schemas TRƯỚC khi CREATE lại
-- Pattern từ quick_260418_zz_cleanup_duplicates.sql nhưng DROP ALL không chỉ duplicate
DO $$
DECLARE r RECORD;
BEGIN
  FOR r IN
    SELECT n.nspname, p.proname, pg_get_function_identity_arguments(p.oid) AS args
    FROM pg_proc p
    JOIN pg_namespace n ON n.oid = p.pronamespace
    WHERE n.nspname IN ('public', 'edoc', 'esto', 'cont', 'iso')
      AND p.proname LIKE 'fn_%'
  LOOP
    EXECUTE format('DROP FUNCTION IF EXISTS %I.%I(%s) CASCADE', r.nspname, r.proname, r.args);
  END LOOP;
END $$;

-- ─── Phần 4: Drop triggers (idempotent) ───
-- [Liệt kê tất cả DROP TRIGGER IF EXISTS ... ON ...; từ các file migration]

-- ─── Phần 5: Tables + Constraints + Indexes ───
-- [Merge từ 000_full_schema + quick_*.sql + 040_signing_schema]
-- Dùng CREATE TABLE IF NOT EXISTS + ALTER TABLE ADD COLUMN IF NOT EXISTS
-- Constraints/indexes: dùng IF NOT EXISTS
-- Trạng thái cuối cùng: bảng staff KHÔNG có column sign_phone (đã drop ở migration 041)

-- ─── Phần 6: Functions (tất cả CREATE OR REPLACE FUNCTION) ───
-- [Merge từ tất cả migration files — public.* + edoc.* + esto.* + cont.* + iso.*]
-- CREATE OR REPLACE — do Phần 3 đã DROP sạch, không còn overload lạc

-- ─── Phần 7: Triggers ───
-- [CREATE TRIGGER ... — sau khi functions đã có]

-- ─── Phần 8: Thông báo hoàn thành ───
DO $$ BEGIN RAISE NOTICE 'Schema v2.0 applied OK'; END $$;
```
