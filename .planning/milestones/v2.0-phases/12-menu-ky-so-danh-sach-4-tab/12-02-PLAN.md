```typescript
'use client';

import { useEffect, useMemo, useRef, useState, useCallback } from 'react';
import { useRouter, useSearchParams } from 'next/navigation';
import {
  App,
  Badge,
  Button,
  Card,
  Modal,
  Skeleton,
  Space,
  Table,
  Tabs,
  Tag,
  Tooltip,
  Typography,
} from 'antd';
import type { ColumnsType, TablePaginationConfig } from 'antd/es/table';
import {
  CloseOutlined,
  DownloadOutlined,
  ReloadOutlined,
  SafetyCertificateOutlined,
} from '@ant-design/icons';
import dayjs from 'dayjs';
import { api } from '@/lib/api';
import { useSigning } from '@/hooks/use-signing';
import { getSocket, SOCKET_EVENTS } from '@/lib/socket';
import type {
  AttachmentType,
  SignCompletedEvent,
  SignFailedEvent,
} from '@/lib/signing/types';

const { Title } = Typography;

// ============================================================================
// Constants
// ============================================================================
type TabKey = 'need_sign' | 'pending' | 'completed' | 'failed';
const TAB_KEYS: TabKey[] = ['need_sign', 'pending', 'completed', 'failed'];
const TAB_LABELS: Record<TabKey, string> = {
  need_sign: 'Cần ký',
  pending: 'Đang xử lý',
  completed: 'Đã ký',
  failed: 'Thất bại',
};
const DEFAULT_PAGE_SIZE = 20;
const PAGE_SIZE_OPTIONS = ['10', '20', '50', '100'];

const DOC_TYPE_LABEL: Record<string, string> = {
  outgoing_doc: 'Văn bản đi',
  drafting_doc: 'Dự thảo',
  handling_doc: 'Hồ sơ công việc',
};
const PROVIDER_LABEL: Record<string, string> = {
  SMARTCA_VNPT: 'VNPT SmartCA',
  MYSIGN_VIETTEL: 'Viettel MySign',
};

// ============================================================================
// Row types (discriminated union theo tab — D-12)
// ============================================================================
interface NeedSignRow {
  attachment_id: number;
  attachment_type: AttachmentType;
  file_name: string;
  doc_id: number;
  doc_type: string;
  doc_label: string;
  doc_number: number;
  doc_notation: string;
  created_at: string;
}
interface TxnRow {
  transaction_id: number;
  provider_code: string;
  provider_name: string;
  attachment_id: number;
  attachment_type: AttachmentType;
  file_name: string;
  doc_id: number;
  doc_type: string;
  doc_label: string;
  status: string;
  error_message: string | null;
  created_at: string;
  completed_at: string | null;
}
interface Counts {
  need_sign: number;
  pending: number;
  completed: number;
  failed: number;
}

// ============================================================================
// Helpers
// ============================================================================
function isValidTab(v: string | null): v is TabKey {
  return !!v && (TAB_KEYS as string[]).includes(v);
}
function parsePositiveInt(v: string | null, fallback: number): number {
  if (!v) return fallback;
  const n = Number(v);
  return Number.isFinite(n) && n > 0 ? n : fallback;
}
function formatDate(iso: string | null): string {
  if (!iso) return '—';
  return dayjs(iso).format('DD/MM/YYYY HH:mm');
}
function docTypeLabel(key: string): string {
  return DOC_TYPE_LABEL[key] ?? key;
}
function providerLabel(code: string, name?: string): string {
  return name || PROVIDER_LABEL[code] || code;
}

// ============================================================================
// Main page
// ============================================================================
export default function DanhSachKySoPage() {
  const { message, modal } = App.useApp();
  const router = useRouter();
  const searchParams = useSearchParams();

  // --- State sync từ URL
  const urlTab = searchParams.get('tab');
  const urlPage = searchParams.get('page');
  const urlPageSize = searchParams.get('pageSize');

  const [activeTab, setActiveTab] = useState<TabKey>(isValidTab(urlTab) ? urlTab : 'need_sign');
  const [page, setPage] = useState<number>(parsePositiveInt(urlPage, 1));
  const [pageSize, setPageSize] = useState<number>(parsePositiveInt(urlPageSize, DEFAULT_PAGE_SIZE));

  // --- Data state
  const [counts, setCounts] = useState<Counts>({ need_sign: 0, pending: 0, completed: 0, failed: 0 });
  const [rows, setRows] = useState<NeedSignRow[] | TxnRow[]>([]);
  const [total, setTotal] = useState<number>(0);
  const [loading, setLoading] = useState<boolean>(true);
  const [initialLoading, setInitialLoading] = useState<boolean>(true);

  // --- Sign modal hook
  const { openSign, renderSignModal } = useSigning();

  // ============================================================================
  // URL sync — Push state khi activeTab / page / pageSize thay đổi
  // ============================================================================
  const syncUrl = useCallback(
    (nextTab: TabKey, nextPage: number, nextPageSize: number) => {
      const params = new URLSearchParams();
      params.set('tab', nextTab);
      params.set('page', String(nextPage));
      params.set('pageSize', String(nextPageSize));
      router.replace(`/ky-so/danh-sach?${params.toString()}`, { scroll: false });
    },
    [router],
  );

  // ============================================================================
  // Fetchers
  // ============================================================================
  const fetchCounts = useCallback(async () => {
    try {
      const res = await api.get('/api/ky-so/danh-sach/counts');
      if (res.data?.success && res.data?.data) {
        setCounts(res.data.data as Counts);
      }
    } catch {
      // Giữ counts cũ — không spam user message error, lần refetch sau sẽ auto-heal
    }
  }, []);

  const fetchList = useCallback(
    async (tab: TabKey, p: number, ps: number) => {
      setLoading(true);
      try {
        const res = await api.get('/api/ky-so/danh-sach', {
          params: { tab, page: p, page_size: ps }, // NOTE: page_size (snake_case) — khớp backend
        });
        if (res.data?.success) {
          setRows(res.data.data ?? []);
          setTotal(res.data.pagination?.total ?? 0);
        } else {
          setRows([]);
          setTotal(0);
          message.error(res.data?.message || 'Không tải được danh sách');
        }
      } catch (err: unknown) {
        const axiosErr = err as { response?: { data?: { message?: string } } };
        message.error(axiosErr?.response?.data?.message || 'Không tải được danh sách');
        setRows([]);
        setTotal(0);
      } finally {
        setLoading(false);
      }
    },
    [message],
  );

  // ============================================================================
  // Mount + state change → refetch
  // ============================================================================
  useEffect(() => {
    (async () => {
      await Promise.all([fetchCounts(), fetchList(activeTab, page, pageSize)]);
      setInitialLoading(false);
    })();
  // eslint-disable-next-line react-hooks/exhaustive-deps
  }, []); // Mount only

  useEffect(() => {
    if (initialLoading) return;
    fetchList(activeTab, page, pageSize);
    syncUrl(activeTab, page, pageSize);
  }, [activeTab, page, pageSize, fetchList, syncUrl, initialLoading]);

  // ============================================================================
  // Socket realtime — SIGN_COMPLETED / SIGN_FAILED
  // ============================================================================
  useEffect(() => {
    const socket = getSocket();
    if (!socket) return;

    const refresh = () => {
      fetchCounts();
      fetchList(activeTab, page, pageSize);
    };
    const onCompleted = (payload: SignCompletedEvent) => {
      // Defense-in-depth T-12-05: nếu event lỗi lọt nhầm từ user khác, refresh
      // chỉ đổi data cho user hiện tại (BE API auto-filter staff_id JWT)
      if (payload?.transaction_id) refresh();
    };
    const onFailed = (payload: SignFailedEvent) => {
      if (payload?.transaction_id) refresh();
    };
    socket.on(SOCKET_EVENTS.SIGN_COMPLETED, onCompleted);
    socket.on(SOCKET_EVENTS.SIGN_FAILED, onFailed);
    return () => {
      socket.off(SOCKET_EVENTS.SIGN_COMPLETED, onCompleted);
      socket.off(SOCKET_EVENTS.SIGN_FAILED, onFailed);
    };
  }, [activeTab, page, pageSize, fetchCounts, fetchList]);

  // ============================================================================
  // Handlers (defined trong Task 2)
  // ============================================================================
  // ... (Task 2 thêm handleCancel, handleDownload, handleSign / handleRetry, columns)

  // ============================================================================
  // Render
  // ============================================================================
  const tabItems = TAB_KEYS.map((key) => ({
    key,
    label: (
      <Space size={8}>
        <span>{TAB_LABELS[key]}</span>
        <Badge
          count={counts[key]}
          showZero
          style={{
            backgroundColor:
              key === 'need_sign' ? '#D97706' :
              key === 'pending' ? '#0891B2' :
              key === 'completed' ? '#059669' :
              '#DC2626',
          }}
        />
      </Space>
    ),
    children: (
      <Table
        rowKey={(row) =>
          'transaction_id' in row
            ? `txn-${row.transaction_id}`
            : `att-${row.attachment_id}`
        }
        dataSource={rows}
        columns={getColumnsForTab(key)}       // ← Task 2 define
        loading={loading}
        pagination={{
          current: page,
          pageSize,
          total,
          showSizeChanger: true,
          pageSizeOptions: PAGE_SIZE_OPTIONS,
          showTotal: (t) => `Tổng ${t} giao dịch`,
          onChange: (p, ps) => {
            setPage(p);
            setPageSize(ps);
          },
        }}
        locale={{
          emptyText: emptyTextForTab(key),  // ← Task 2 define
        }}
      />
    ),
  }));

  if (initialLoading) {
    return (
      <div>
        <div className="page-header">
          <Title level={3} className="page-title">
            <SafetyCertificateOutlined /> Danh sách ký số
          </Title>
        </div>
        <Card className="page-card">
          <Skeleton active paragraph={{ rows: 8 }} />
        </Card>
      </div>
    );
  }

  return (
    <>
      <div className="page-header">
        <Title level={3} className="page-title">
          <SafetyCertificateOutlined /> Danh sách ký số
        </Title>
      </div>
      <Card className="page-card">
        <Tabs
          activeKey={activeTab}
          onChange={(k) => {
            if (isValidTab(k)) {
              setActiveTab(k);
              setPage(1); // Tab switch → reset page 1, giữ pageSize (D-11)
            }
          }}
          items={tabItems}
        />
      </Card>
      {renderSignModal()}
    </>
  );
}
```

```typescript
  /** Mở modal ký cho row Cần ký hoặc Thất bại (ký lại) — dùng useSigning hook */
  const handleSign = useCallback(
    (row: NeedSignRow | TxnRow) => {
      openSign({
        attachment: { id: row.attachment_id, file_name: row.file_name },
        attachmentType: row.attachment_type,
        docId: row.doc_id,
        onSuccess: () => {
          // Refresh list + counts sau khi ký thành công
          fetchCounts();
          fetchList(activeTab, page, pageSize);
        },
      });
    },
    [openSign, fetchCounts, fetchList, activeTab, page, pageSize],
  );

  /** Hủy transaction pending */
  const handleCancel = useCallback(
    (row: TxnRow) => {
      modal.confirm({
        title: 'Hủy giao dịch ký số',
        content: `Bạn có chắc muốn hủy giao dịch ký cho file "${row.file_name}"?`,
        okText: 'Hủy giao dịch',
        okType: 'danger',
        cancelText: 'Đóng',
        maskClosable: false,
        onOk: async () => {
          try {
            const res = await api.post(`/api/ky-so/sign/${row.transaction_id}/cancel`);
            if (res.data?.success) {
              message.success('Đã hủy giao dịch ký số');
              await Promise.all([fetchCounts(), fetchList(activeTab, page, pageSize)]);
            } else {
              message.error(res.data?.message || 'Không thể hủy giao dịch');
            }
          } catch (err: unknown) {
            const axiosErr = err as { response?: { data?: { message?: string } } };
            message.error(axiosErr?.response?.data?.message || 'Không thể hủy giao dịch');
          }
        },
      });
    },
    [modal, message, fetchCounts, fetchList, activeTab, page, pageSize],
  );

  /** Tải file đã ký — gọi endpoint Plan 12-01 */
  const handleDownload = useCallback(
    async (row: TxnRow) => {
      try {
        const res = await api.get(`/api/ky-so/sign/${row.transaction_id}/download`);
        const url = res.data?.data?.url;
        if (!url) {
          message.error('Không lấy được link tải file');
          return;
        }
        // Mở tab mới — browser tự trigger download
        window.open(url, '_blank', 'noopener,noreferrer');
      } catch (err: unknown) {
        const axiosErr = err as { response?: { data?: { message?: string }; status?: number } };
        const status = axiosErr?.response?.status;
        const msg =
          axiosErr?.response?.data?.message ||
          (status === 404 ? 'File đã ký chưa sẵn sàng' : 'Không tải được file');
        message.error(msg);
      }
    },
    [message],
  );
```
