<objective>
Implement the synchronous part of the sign flow — POST /api/ky-so/sign returns `{ transaction_id }` fast (< 1s). All the slow stuff (polling provider, embedding signature, uploading signed PDF, emitting Socket event) happens in Plan 04's worker.

Purpose: This is the user-facing entry point. When user clicks "Ký số" on a detail page, frontend hits this endpoint and gets an immediate transaction_id. Plan 06's SignModal then polls GET /:id or waits for Socket event. If the user closes the browser, the worker still finishes the sign and updates DB + emits notification.

Output: 1 placeholder store module + 1 route file with 3 endpoints + server.ts mount line.

The worker (Plan 04) needs the placeholder PDF buffer to embed signature. We cannot keep it in memory between POST /sign and the delayed worker job (backend may restart). Redis unsuitable for ~200KB+ PDFs. Use MinIO with a dedicated prefix, delete after finalize.

```typescript
/**
 * Ephemeral MinIO store for placeholder PDFs between POST /sign and worker finalize.
 *
 * Why MinIO (not Redis): placeholder PDFs can be 200KB+. Redis is for small
 * job payloads, not blob data. MinIO already available + already stores docs.
 *
 * Lifecycle:
 *   1. POST /sign writes placeholder → 'signing-placeholders/{txnId}.pdf'
 *   2. Worker getPlaceholder(txnId) to retrieve buffer
 *   3. Worker removePlaceholder(txnId) after finalize/fail/cancel
 *
 * Cleanup on failure: worker unconditionally removes on all terminal paths.
 * If worker crashes mid-flow, placeholder becomes orphan — deploy should add
 * lifecycle policy on MinIO prefix 'signing-placeholders/' to auto-expire
 * after 24h. Documented in Plan 11-08 deployment hardening.
 */

import { minioClient } from '../minio/client.js';

const BUCKET = process.env.MINIO_BUCKET || 'documents';
export const PLACEHOLDER_PREFIX = 'signing-placeholders';

function keyFor(txnId: number): string {
  return `${PLACEHOLDER_PREFIX}/${txnId}.pdf`;
}

async function streamToBuffer(stream: NodeJS.ReadableStream): Promise<Buffer> {
  const chunks: Buffer[] = [];
  for await (const chunk of stream) {
    chunks.push(Buffer.isBuffer(chunk) ? chunk : Buffer.from(chunk as string));
  }
  return Buffer.concat(chunks);
}

export async function putPlaceholder(txnId: number, buffer: Buffer): Promise<void> {
  await minioClient.putObject(BUCKET, keyFor(txnId), buffer, buffer.length, {
    'Content-Type': 'application/pdf',
  });
}

export async function getPlaceholder(txnId: number): Promise<Buffer> {
  const stream = await minioClient.getObject(BUCKET, keyFor(txnId));
  return streamToBuffer(stream);
}

export async function removePlaceholder(txnId: number): Promise<void> {
  try {
    await minioClient.removeObject(BUCKET, keyFor(txnId));
  } catch {
    // Not fatal — object may already be gone
  }
}

/** Used by worker to download the ORIGINAL file from MinIO (not the placeholder). */
export async function downloadOriginalPdf(originalKey: string): Promise<Buffer> {
  const stream = await minioClient.getObject(BUCKET, originalKey);
  return streamToBuffer(stream);
}
```

```typescript
/**
 * Route: /api/ky-so/sign — Async sign flow entry point (Phase 11).
 *
 * 3 endpoints (all require authenticate, no requireRoles — any user can sign
 * if they have permission on the specific attachment):
 *   POST   /            — Bắt đầu flow ký (< 1s response, enqueue poll job)
 *   POST   /:id/cancel  — Hủy transaction đang pending
 *   GET    /:id         — Xem trạng thái transaction (cho frontend fallback polling)
 *
 * SECURITY:
 *   - staffId luôn từ req.user.staffId (JWT) — NEVER body (T-11-01 Tampering)
 *   - attachmentSignRepository.canSign() enforces owner/signer permission (T-11-02)
 *   - GET /:id owner-only — staff chỉ đọc transaction của chính mình (T-11-03)
 *
 * Performance target: POST / completes in < 1s:
 *   - DB lookups: ~50ms
 *   - MinIO download (10-500KB PDF): ~100-500ms
 *   - prepareSignPdf (pure JS, SHA256): ~50ms
 *   - provider.signHash (HTTP call to Viettel/VNPT): ~300-800ms
 *   - signTransaction.create + setProviderTxn + queue.add: ~100ms
 *   Total worst case: ~1.5s (acceptable). Typical: 500-700ms.
 */

import { Router, type Request, type Response } from 'express';
import type { AuthRequest } from '../middleware/auth.js';
import { handleDbError } from '../lib/error-handler.js';
import { signTransactionRepository, type SignAttachmentType } from '../repositories/sign-transaction.repository.js';
import { attachmentSignRepository } from '../repositories/attachment-sign.repository.js';
import { staffSigningConfigRepository } from '../repositories/staff-signing-config.repository.js';
import { getActiveProviderWithCredentials } from '../services/signing/providers/provider-factory.js';
import { prepareSignPdf } from '../services/signing/pdf-signer.js';
import { downloadOriginalPdf, putPlaceholder, removePlaceholder } from '../lib/signing/placeholder-store.js';
import { enqueuePollSignStatus, cancelPollJobsForTransaction } from '../lib/queue/signing-queue.js';
import { isValidAttachmentType } from '../lib/signing/sign-helpers.js';

const router = Router();

// Map attachment_type → doc_type stored in sign_transactions (SP expects this link)
const ATTACHMENT_TO_DOC_TYPE: Record<SignAttachmentType, string> = {
  incoming: 'incoming_doc',
  outgoing: 'outgoing_doc',
  drafting: 'drafting_doc',
  handling: 'handling_doc',
};

// ============================================================================
// POST / — Start sign flow
// Body: { attachment_id, attachment_type, doc_id?, sign_reason?, sign_location? }
// ============================================================================
router.post('/', async (req: Request, res: Response) => {
  const startedAt = Date.now();
  try {
    const { staffId } = (req as AuthRequest).user;
    const { attachment_id, attachment_type, doc_id, sign_reason, sign_location } = req.body ?? {};

    // ---- Validate input
    if (!Number.isFinite(Number(attachment_id)) || Number(attachment_id) <= 0) {
      return res.status(400).json({ success: false, message: 'attachment_id không hợp lệ' });
    }
    if (!isValidAttachmentType(attachment_type)) {
      return res.status(400).json({ success: false, message: 'attachment_type phải là incoming/outgoing/drafting/handling' });
    }
    if (attachment_type === 'incoming') {
      return res.status(400).json({ success: false, message: 'Không được ký số văn bản đến' });
    }

    // ---- 1. Check active provider
    const active = await getActiveProviderWithCredentials();
    if (!active) {
      return res.status(400).json({
        success: false,
        message: 'Hệ thống chưa cấu hình provider ký số. Vui lòng liên hệ Quản trị viên.',
      });
    }
    const providerCode = active.provider.code;

    // ---- 2. Check user has verified staff_signing_config
    const userConfig = await staffSigningConfigRepository.get(staffId, providerCode);
    if (!userConfig) {
      return res.status(400).json({
        success: false,
        message: 'Bạn chưa cấu hình tài khoản ký số. Vui lòng vào "Tài khoản ký số cá nhân" để cấu hình.',
      });
    }
    if (!userConfig.is_verified) {
      return res.status(400).json({
        success: false,
        message: 'Tài khoản ký số chưa được xác thực. Vui lòng bấm "Kiểm tra" trong trang Tài khoản ký số cá nhân.',
      });
    }

    // ---- 3. Permission + get file_path
    const canSign = await attachmentSignRepository.canSign(
      Number(attachment_id),
      attachment_type,
      staffId,
    );
    if (!canSign.can_sign) {
      return res.status(403).json({ success: false, message: canSign.reason ?? 'Không có quyền ký' });
    }
    if (!canSign.file_path) {
      return res.status(404).json({ success: false, message: 'Không tìm thấy đường dẫn file' });
    }
    const fileName = canSign.file_name ?? 'document.pdf';

    // ---- 4. Basic PDF mime check (extension)
    if (!fileName.toLowerCase().endsWith('.pdf')) {
      return res.status(400).json({
        success: false,
        message: 'Chỉ hỗ trợ ký file PDF. File này định dạng không hợp lệ.',
      });
    }

    // ---- 5. Download PDF from MinIO + prepare placeholder + hash
    let pdfBuffer: Buffer;
    try {
      pdfBuffer = await downloadOriginalPdf(canSign.file_path);
    } catch (err: unknown) {
      return res.status(500).json({
        success: false,
        message: 'Không thể tải file PDF từ MinIO: ' + (err instanceof Error ? err.message : String(err)),
      });
    }

    let hashResult;
    try {
      hashResult = prepareSignPdf(pdfBuffer, {
        reason: (typeof sign_reason === 'string' && sign_reason.trim()) ? sign_reason.trim().slice(0, 200) : 'Ký phê duyệt văn bản',
        location: (typeof sign_location === 'string' && sign_location.trim()) ? sign_location.trim().slice(0, 200) : 'Vietnam',
        name: 'E-Office signer',
      });
    } catch (err: unknown) {
      return res.status(400).json({
        success: false,
        message: 'File PDF không hợp lệ hoặc không tương thích ký số: ' + (err instanceof Error ? err.message : String(err)),
      });
    }

    // ---- 6. Create DB transaction FIRST (so we have txnId for placeholder key)
    const createResult = await signTransactionRepository.create({
      staffId,
      providerCode,
      attachmentId: Number(attachment_id),
      attachmentType: attachment_type,
      docId: doc_id ? Number(doc_id) : null,
      docType: ATTACHMENT_TO_DOC_TYPE[attachment_type],
      fileHashSha256: hashResult.hash,
    });
    if (!createResult.success) {
      return res.status(400).json({ success: false, message: createResult.message });
    }
    const txnId = createResult.id;

    // ---- 7. Store placeholder PDF for worker to retrieve later
    try {
      await putPlaceholder(txnId, hashResult.placeholderPdf);
    } catch (err: unknown) {
      await signTransactionRepository.updateStatus(txnId, 'failed', 'Không thể lưu placeholder PDF: ' + String(err));
      return res.status(500).json({ success: false, message: 'Lỗi lưu placeholder' });
    }

    // ---- 8. Call provider.signHash → get providerTxnId (OTP pushed to user's app)
    let signHashResult;
    try {
      signHashResult = await active.provider.signHash(
        active.credentials,
        { userId: userConfig.user_id, credentialId: userConfig.credential_id },
        {
          hashHex: hashResult.hash,
          documentName: fileName,
          documentId: String(txnId),
        },
      );
    } catch (err: unknown) {
      const msg = err instanceof Error ? err.message : String(err);
      await signTransactionRepository.updateStatus(txnId, 'failed', 'Provider reject: ' + msg.slice(0, 500));
      await removePlaceholder(txnId);
      return res.status(502).json({
        success: false,
        message: 'Provider từ chối yêu cầu ký: ' + msg,
      });
    }

    // ---- 9. Save provider_txn_id + enqueue poll job (delay 5s to give user time)
    await signTransactionRepository.setProviderTxn(txnId, signHashResult.providerTransactionId);
    await enqueuePollSignStatus({
      signTransactionId: txnId,
      providerTransactionId: signHashResult.providerTransactionId,
      placeholderPdfKey: `signing-placeholders/${txnId}.pdf`,
      attempt: 1,
    });

    // ---- 10. Return fast
    const elapsedMs = Date.now() - startedAt;
    return res.status(201).json({
      success: true,
      data: {
        transaction_id: txnId,
        provider_transaction_id: signHashResult.providerTransactionId,
        provider_code: providerCode,
        provider_message: signHashResult.message ?? null,
        elapsed_ms: elapsedMs,
      },
      message: 'Yêu cầu ký đã được gửi. Vui lòng xác nhận OTP trên ứng dụng di động.',
    });
  } catch (error) {
    handleDbError(error, res);
  }
});

// ============================================================================
// POST /:id/cancel — Hủy transaction
// ============================================================================
router.post('/:id/cancel', async (req: Request, res: Response) => {
  try {
    const { staffId } = (req as AuthRequest).user;
    const id = Number(req.params.id);
    if (!Number.isFinite(id)) {
      return res.status(400).json({ success: false, message: 'ID không hợp lệ' });
    }

    // Owner check
    const txn = await signTransactionRepository.getById(id);
    if (!txn) {
      return res.status(404).json({ success: false, message: 'Không tìm thấy giao dịch' });
    }
    if (txn.staff_id !== staffId) {
      return res.status(403).json({ success: false, message: 'Bạn không có quyền hủy giao dịch này' });
    }
    if (txn.status !== 'pending') {
      return res.status(400).json({
        success: false,
        message: `Giao dịch không thể hủy (trạng thái hiện tại: ${txn.status})`,
      });
    }

    // Mark DB cancelled
    const result = await signTransactionRepository.updateStatus(id, 'cancelled', 'Hủy bởi người dùng');
    if (!result.success) {
      return res.status(400).json({ success: false, message: result.message });
    }

    // Remove delayed queue jobs (best-effort — worker also DB-checks status)
    try { await cancelPollJobsForTransaction(id); } catch { /* non-fatal */ }
    await removePlaceholder(id);

    return res.json({ success: true, data: { transaction_id: id }, message: 'Đã hủy giao dịch' });
  } catch (error) {
    handleDbError(error, res);
  }
});

// ============================================================================
// GET /:id — Xem trạng thái transaction (owner only)
// ============================================================================
router.get('/:id', async (req: Request, res: Response) => {
  try {
    const { staffId } = (req as AuthRequest).user;
    const id = Number(req.params.id);
    if (!Number.isFinite(id)) {
      return res.status(400).json({ success: false, message: 'ID không hợp lệ' });
    }

    const txn = await signTransactionRepository.getById(id);
    if (!txn) {
      return res.status(404).json({ success: false, message: 'Không tìm thấy giao dịch' });
    }
    if (txn.staff_id !== staffId) {
      return res.status(403).json({ success: false, message: 'Không có quyền xem giao dịch này' });
    }

    return res.json({
      success: true,
      data: {
        id: txn.id,
        status: txn.status,
        provider_code: txn.provider_code,
        provider_txn_id: txn.provider_txn_id,
        attachment_id: txn.attachment_id,
        attachment_type: txn.attachment_type,
        error_message: txn.error_message,
        retry_count: txn.retry_count,
        created_at: txn.created_at,
        started_at: txn.started_at,
        completed_at: txn.completed_at,
        expires_at: txn.expires_at,
        signed_file_path: txn.signed_file_path,
      },
    });
  } catch (error) {
    handleDbError(error, res);
  }
});

export default router;
```
