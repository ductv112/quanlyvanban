**SP 1: fn_sign_need_list_by_staff** — "Cần ký" tab.
Signature: `(p_staff_id INT, p_page INT, p_page_size INT)` returns `TABLE(attachment_id BIGINT, attachment_type VARCHAR, file_name VARCHAR, doc_id BIGINT, doc_type VARCHAR, doc_label TEXT, doc_number INT, doc_notation VARCHAR, created_at TIMESTAMPTZ, total_count BIGINT)`.

Logic: UNION ALL 3 queries (outgoing / drafting / handling). For each:
- LEFT JOIN parent doc table
- Filter: attachment.is_ca = FALSE AND attachment not referenced in pending sign_transactions
- Filter: user can sign (same logic as fn_attachment_can_sign):
  - outgoing/drafting: `LOWER(UNACCENT(od.signer)) = LOWER(UNACCENT(staff.full_name))` OR `od.created_by = p_staff_id` OR staff has admin role
  - handling: `hd.signer = p_staff_id`
- doc_label: 'VB đi số {number}/{notation}' / 'VB dự thảo số {number}/{notation}' / 'HSCV: {name}'

Skeleton:
```sql
CREATE OR REPLACE FUNCTION edoc.fn_sign_need_list_by_staff(
  p_staff_id INT, p_page INT, p_page_size INT
)
RETURNS TABLE(
  attachment_id BIGINT, attachment_type VARCHAR(20),
  file_name VARCHAR, doc_id BIGINT, doc_type VARCHAR(20),
  doc_label TEXT, doc_number INT, doc_notation VARCHAR,
  created_at TIMESTAMPTZ, total_count BIGINT
)
LANGUAGE plpgsql STABLE AS $$
DECLARE
  v_staff_name VARCHAR;
  v_offset INT := ((COALESCE(p_page,1) - 1) * COALESCE(p_page_size,20));
  v_limit  INT := COALESCE(p_page_size, 20);
BEGIN
  SELECT full_name INTO v_staff_name FROM public.staff WHERE id = p_staff_id;

  RETURN QUERY
  WITH combined AS (
    -- Outgoing docs
    SELECT aod.id                AS attachment_id,
           'outgoing'::VARCHAR   AS attachment_type,
           aod.file_name,
           od.id                 AS doc_id,
           'outgoing_doc'::VARCHAR AS doc_type,
           ('VB đi số ' || COALESCE(od.number::TEXT, '?') || ' — ' || COALESCE(od.notation,'?'))::TEXT AS doc_label,
           od.number AS doc_number,
           od.notation AS doc_notation,
           aod.created_at
      FROM edoc.attachment_outgoing_docs aod
      JOIN edoc.outgoing_docs od ON od.id = aod.outgoing_doc_id
     WHERE aod.is_ca = FALSE
       AND LOWER(aod.file_name) LIKE '%.pdf'
       AND NOT EXISTS (
         SELECT 1 FROM edoc.sign_transactions st
          WHERE st.attachment_id = aod.id
            AND st.attachment_type = 'outgoing'
            AND st.status = 'pending'
       )
       AND (
         od.created_by = p_staff_id
         OR LOWER(UNACCENT(COALESCE(od.signer,''))) = LOWER(UNACCENT(v_staff_name))
       )
    UNION ALL
    -- Drafting docs
    SELECT add2.id, 'drafting'::VARCHAR, add2.file_name,
           dd.id, 'drafting_doc'::VARCHAR,
           ('VB dự thảo số ' || COALESCE(dd.number::TEXT,'?') || ' — ' || COALESCE(dd.notation,'?'))::TEXT,
           dd.number, dd.notation, add2.created_at
      FROM edoc.attachment_drafting_docs add2
      JOIN edoc.drafting_docs dd ON dd.id = add2.drafting_doc_id
     WHERE add2.is_ca = FALSE
       AND LOWER(add2.file_name) LIKE '%.pdf'
       AND NOT EXISTS (
         SELECT 1 FROM edoc.sign_transactions st
          WHERE st.attachment_id = add2.id AND st.attachment_type = 'drafting' AND st.status = 'pending'
       )
       AND (
         dd.created_by = p_staff_id
         OR LOWER(UNACCENT(COALESCE(dd.signer,''))) = LOWER(UNACCENT(v_staff_name))
       )
    UNION ALL
    -- Handling docs (HSCV)
    SELECT ahd.id, 'handling'::VARCHAR, ahd.file_name,
           hd.id, 'handling_doc'::VARCHAR,
           ('HSCV: ' || COALESCE(hd.name, 'không tên'))::TEXT,
           NULL::INT, NULL::VARCHAR, ahd.created_at
      FROM edoc.attachment_handling_docs ahd
      JOIN edoc.handling_docs hd ON hd.id = ahd.handling_doc_id
     WHERE COALESCE(ahd.is_ca, FALSE) = FALSE
       AND LOWER(ahd.file_name) LIKE '%.pdf'
       AND NOT EXISTS (
         SELECT 1 FROM edoc.sign_transactions st
          WHERE st.attachment_id = ahd.id AND st.attachment_type = 'handling' AND st.status = 'pending'
       )
       AND hd.signer = p_staff_id
       AND hd.status IN (2, 3)
  )
  SELECT c.attachment_id, c.attachment_type, c.file_name,
         c.doc_id, c.doc_type, c.doc_label, c.doc_number, c.doc_notation, c.created_at,
         COUNT(*) OVER()::BIGINT AS total_count
    FROM combined c
   ORDER BY c.created_at DESC
   LIMIT v_limit OFFSET v_offset;
END;
$$;
```

```typescript
/**
 * Route: /api/ky-so/danh-sach — list signatures for current user (Phase 11 Plan 05).
 *
 * 2 endpoints (authenticate only; no admin role required):
 *   GET  /        — paginated list by tab
 *   GET  /counts  — 4 counts for sidebar badges (need_sign + 3 sign_transactions statuses)
 *
 * Tab values:
 *   need_sign — VB/HSCV user can sign, chưa ký (SP fn_sign_need_list_by_staff)
 *   pending   — sign_transactions WHERE status='pending' AND staff_id=me
 *   completed — sign_transactions WHERE status='completed' AND staff_id=me
 *   failed    — sign_transactions WHERE status IN ('failed','expired','cancelled') AND staff_id=me
 *
 * staffId always from JWT (T-11-03 Info Disclosure mitigation) — never from query.
 */

import { Router, type Request, type Response } from 'express';
import type { AuthRequest } from '../middleware/auth.js';
import { handleDbError } from '../lib/error-handler.js';
import { attachmentSignRepository, type SignListTab } from '../repositories/attachment-sign.repository.js';

const router = Router();

const VALID_TABS = ['need_sign', 'pending', 'completed', 'failed'] as const;
type ValidTab = typeof VALID_TABS[number];

function isValidTab(v: unknown): v is ValidTab {
  return typeof v === 'string' && (VALID_TABS as readonly string[]).includes(v);
}

// ============================================================================
// GET /counts — badge counts for sidebar
// ============================================================================
router.get('/counts', async (req: Request, res: Response) => {
  try {
    const { staffId } = (req as AuthRequest).user;
    const [txnCounts, needCount] = await Promise.all([
      attachmentSignRepository.countByStaff(staffId),
      attachmentSignRepository.needCountByStaff(staffId),
    ]);
    return res.json({
      success: true,
      data: {
        need_sign: needCount,
        pending: Number(txnCounts.pending_count ?? 0),
        completed: Number(txnCounts.completed_count ?? 0),
        failed: Number(txnCounts.failed_count ?? 0),
      },
    });
  } catch (error) {
    handleDbError(error, res);
  }
});

// ============================================================================
// GET / — paginated list by tab
// Query: ?tab=need_sign|pending|completed|failed, page=1, page_size=20
// ============================================================================
router.get('/', async (req: Request, res: Response) => {
  try {
    const { staffId } = (req as AuthRequest).user;
    const tabRaw = String(req.query.tab ?? 'need_sign');
    if (!isValidTab(tabRaw)) {
      return res.status(400).json({
        success: false,
        message: `tab phải là một trong: ${VALID_TABS.join(', ')}`,
      });
    }
    const page = Math.max(1, Number(req.query.page) || 1);
    const pageSize = Math.min(100, Math.max(1, Number(req.query.page_size) || 20));

    if (tabRaw === 'need_sign') {
      const rows = await attachmentSignRepository.needListByStaff(staffId, page, pageSize);
      const total = Number(rows[0]?.total_count ?? 0);
      return res.json({
        success: true,
        data: rows.map((r) => ({
          attachment_id: r.attachment_id,
          attachment_type: r.attachment_type,
          file_name: r.file_name,
          doc_id: r.doc_id,
          doc_type: r.doc_type,
          doc_label: r.doc_label,
          doc_number: r.doc_number,
          doc_notation: r.doc_notation,
          created_at: r.created_at,
        })),
        pagination: { total, page, pageSize },
      });
    }

    // Transaction tabs
    const tab: SignListTab = tabRaw as SignListTab; // pending/completed/failed
    const rows = await attachmentSignRepository.listByStaff(staffId, tab, page, pageSize);
    const total = Number(rows[0]?.total_count ?? 0);
    return res.json({
      success: true,
      data: rows.map((r) => ({
        transaction_id: r.id,
        provider_code: r.provider_code,
        provider_name: r.provider_name,
        attachment_id: r.attachment_id,
        attachment_type: r.attachment_type,
        file_name: r.file_name,
        doc_id: r.doc_id,
        doc_type: r.doc_type,
        doc_label: r.doc_label,
        status: r.status,
        error_message: r.error_message,
        created_at: r.created_at,
        completed_at: r.completed_at,
      })),
      pagination: { total, page, pageSize },
    });
  } catch (error) {
    handleDbError(error, res);
  }
});

export default router;
```
