```typescript
/**
 * ioredis singleton for BullMQ queue + worker.
 *
 * CRITICAL: maxRetriesPerRequest MUST be null — BullMQ requirement.
 * Without it, ioredis throws on reconnect and kills the worker.
 *
 * Reuse the SAME connection for Queue and Worker (sharing connection is safe
 * per BullMQ docs — Worker uses BLPOP which blocks the connection).
 *
 * Wait — actually BullMQ v5 requires SEPARATE connections for Queue vs Worker
 * because Worker blocks the connection. We export a factory that returns
 * a NEW ioredis per call — caller keeps singleton scope as needed.
 */

import IORedis from 'ioredis';
import type { Redis } from 'ioredis';

let defaultConnection: Redis | null = null;

function parseRedisUrl(): { host: string; port: number; password?: string } {
  const url = process.env.REDIS_URL;
  if (url) {
    const u = new URL(url);
    return {
      host: u.hostname,
      port: u.port ? Number(u.port) : 6379,
      password: u.password || undefined,
    };
  }
  return {
    host: process.env.REDIS_HOST || 'localhost',
    port: process.env.REDIS_PORT ? Number(process.env.REDIS_PORT) : 6379,
    password: process.env.REDIS_PASSWORD || undefined,
  };
}

/**
 * Get/reuse default Redis connection for Queue (producer side).
 * For Worker, call `createRedisConnection()` to get a fresh one (blocking connection).
 */
export function getRedisConnection(): Redis {
  if (defaultConnection && defaultConnection.status !== 'end') {
    return defaultConnection;
  }
  const cfg = parseRedisUrl();
  defaultConnection = new IORedis({
    host: cfg.host,
    port: cfg.port,
    password: cfg.password,
    maxRetriesPerRequest: null,
    enableReadyCheck: false,
    lazyConnect: false,
  });
  return defaultConnection;
}

/**
 * Create a NEW Redis connection (worker side — do NOT share with Queue).
 * Caller is responsible for lifecycle (worker.close() will close this).
 */
export function createRedisConnection(): Redis {
  const cfg = parseRedisUrl();
  return new IORedis({
    host: cfg.host,
    port: cfg.port,
    password: cfg.password,
    maxRetriesPerRequest: null,
    enableReadyCheck: false,
  });
}

/** Graceful shutdown helper. */
export async function closeRedisConnection(): Promise<void> {
  if (defaultConnection) {
    await defaultConnection.quit();
    defaultConnection = null;
  }
}
```

**File 1 — signing-queue.ts** (~75 lines):

```typescript
/**
 * BullMQ Queue for async sign flow.
 *
 * Producer side only — no Worker here. Plan 04 adds the Worker that consumes
 * `poll-sign-status` jobs.
 *
 * Flow (happy path):
 *   1. Route POST /api/ky-so/sign creates sign_transactions row + calls provider.signHash
 *   2. Route calls enqueuePollSignStatus(txnId, providerTxnId, placeholderKey, attempt=1) with delay=5s
 *   3. Worker consumes job → polls provider → if still pending, re-enqueues attempt+1
 *   4. If provider done → worker embeds signature, uploads to MinIO, updates DB, emits Socket event
 *   5. If provider failed/expired → worker marks transaction failed + emits Socket event
 *   6. If attempt > MAX_POLL_ATTEMPTS → worker marks expired
 */

import { Queue } from 'bullmq';
import { getRedisConnection } from './redis-connection.js';
import {
  SIGNING_QUEUE_NAME,
  POLL_SIGN_STATUS,
  POLL_INTERVAL_MS,
  JOB_REMOVE_ON_COMPLETE,
  JOB_REMOVE_ON_FAIL,
  type PollSignStatusJob,
} from './types.js';
import pino from 'pino';

const logger = pino({ name: 'signing-queue' });

export { SIGNING_QUEUE_NAME };

/**
 * Module-level singleton. Lazy-initialized on first access so backend without
 * Redis running (tests, CI) doesn't crash at import time — only when sign
 * feature is actually used.
 */
let _queue: Queue<PollSignStatusJob> | null = null;

export function getSigningQueue(): Queue<PollSignStatusJob> {
  if (_queue) return _queue;
  _queue = new Queue<PollSignStatusJob>(SIGNING_QUEUE_NAME, {
    connection: getRedisConnection(),
    defaultJobOptions: {
      removeOnComplete: { count: JOB_REMOVE_ON_COMPLETE },
      removeOnFail:     { count: JOB_REMOVE_ON_FAIL },
      attempts: 1, // We implement retry manually by re-enqueuing (finer control)
    },
  });
  logger.info({ queue: SIGNING_QUEUE_NAME }, 'BullMQ signing queue initialized');
  return _queue;
}

/**
 * Enqueue a `poll-sign-status` job.
 *
 * @param delayMs — milliseconds to wait before worker picks up (default 5000).
 *                  For initial enqueue from route: 5000ms (give provider time
 *                  to push OTP to user). For re-queues from worker: 5000ms too.
 */
export async function enqueuePollSignStatus(
  payload: PollSignStatusJob,
  delayMs: number = POLL_INTERVAL_MS,
): Promise<string> {
  const queue = getSigningQueue();
  // jobId ensures idempotency if same payload enqueued twice (dedupe inside BullMQ)
  const jobId = `poll-${payload.signTransactionId}-${payload.attempt}`;
  const job = await queue.add(POLL_SIGN_STATUS, payload, {
    jobId,
    delay: delayMs,
  });
  logger.debug(
    { jobId: job.id, signTransactionId: payload.signTransactionId, attempt: payload.attempt, delayMs },
    'Enqueued poll-sign-status job',
  );
  return job.id ?? jobId;
}

/**
 * Cancel pending poll jobs for a given transaction (when user clicks "Hủy ký").
 * Removes all jobs matching the signTransactionId prefix — any attempt.
 *
 * NOTE: Active jobs (currently executing) cannot be cancelled mid-flight;
 * worker itself checks DB status at start of handler and skips if cancelled.
 */
export async function cancelPollJobsForTransaction(signTransactionId: number): Promise<number> {
  const queue = getSigningQueue();
  const delayed = await queue.getDelayed();
  let removed = 0;
  for (const job of delayed) {
    if (job.data.signTransactionId === signTransactionId) {
      try {
        await job.remove();
        removed++;
      } catch {
        // Job moved to active during iteration — let worker handle cancellation via DB check
      }
    }
  }
  return removed;
}

/** Graceful shutdown (call from server.ts SIGTERM handler). */
export async function closeSigningQueue(): Promise<void> {
  if (_queue) {
    await _queue.close();
    _queue = null;
  }
}
```
